CREATE TABLE users(
  usrID INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255) NOT NULL UNIQUE,
  role VARCHAR(255) NOT NULL,
  pass VARCHAR(255) NOT NULL
);

CREATE TABLE track_keys(
  keyID INT AUTO_INCREMENT PRIMARY KEY,
  keyName VARCHAR(255),
  lastPos VARCHAR(255),
  triggered BOOLEAN,
  holder VARCHAR(255)
);

CREATE TABLE usr_key(
  usrID INT PRIMARY KEY,
  keyID INT NOT NULL,
  position VARCHAR(255) NOT NULL
);
